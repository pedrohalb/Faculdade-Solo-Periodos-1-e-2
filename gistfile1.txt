struct lista_veiculos {
  string modelo;
  string marca;
  string tipo;
  int ano;
  string km;
  string potencia_do_motor;
  string combustivel;
  string cambio;
  string direcao;
  string cor;
  string portas;
  string placa;
};
lista_veiculos *p[50];

int busca1 = 0;

int busca_placa(lista_veiculos *p[], string placa, int *tam) {
  int j = 0;
  while (j < *tam) {
    if (p[j]->placa == placa) {
      busca1 = j + 1;
      j = *tam;
    } else {
      j++;
    }
  }
  return busca1;
}

int remove(lista_veiculos *p[], int *tam, string placa) {
  int r, i = 0, indice;
  indice = busca_placa(p, placa, tam);
  if (indice != 0) {
    cout << "carro encontrado, deseja removÃª-lo? (s-1/n-0) ";
    cin >> r;
    if (r == 0) {
      return 0;
    } else {
      for (int j = busca1; j < *tam; j++) {
        if (j < *tam) {
          p[j] = p[j + 1];
        } else {
          p[j] = NULL;
        }
      }
      *tam = *tam - 1;
    }
  }
  return 1;
}
